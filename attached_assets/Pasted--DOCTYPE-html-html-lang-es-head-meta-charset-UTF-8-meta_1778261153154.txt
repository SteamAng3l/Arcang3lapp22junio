function detectCategory(text) {
  const normalized = text.toLowerCase();
  for (const category of categories) {
    for (const keyword of category.keywords) {
      if (normalized.includes(keyword)) {
        return category.name;
      }
    }
  }
  return "esperanza";
}

function getResponseMessage(category) {
  const messages = {
    ansiedad: "Dios invita a dejar la ansiedad en sus manos. Respira, ora y descansa en su cuidado.",

    sabiduria: "Dios puede dar dirección y sabiduría cuando no sabes qué hacer."
  };

  return messages[category] || messages.esperanza;

app.post("/api/verse", (req, res) => {
  const { problem } = req.body;

  if (!problem || problem.trim() === "") {
    return res.status(400).json({
      error: "Por favor escribe un problema o situación."
    });
  }

  const category = detectCategory(problem);

  db.get(
    "SELECT category, verse_reference, verse_text FROM verses WHERE category = ? ORDER BY RANDOM() LIMIT 1",
    [category],
    (err, row) => {
      if (err) {
        console.error("Error consultando la base de datos:", err.message);
        return res.status(500).json({ error: "Error interno del servidor." });
      }

      if (!row) {
        return res.status(404).json({ error: "No se encontró un versículo para esa categoría." });
      }

      return res.json({
        detected_category: row.category,
        message: getResponseMessage(row.category),
        verse_reference: row.verse_reference,
        verse_text: row.verse_text
      });
    }
  );
});

app.get("/api/categories", (req, res) => {
  db.all("SELECT DISTINCT category FROM verses ORDER BY category ASC", [], (err, rows) => {
    if (err) {
      return res.status(500).json({ error: "No se pudieron obtener las categorías." });
    }
    res.json(rows);
  });
});
